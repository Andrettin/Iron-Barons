diplomat = {
	name = "Diplomat"
	type = expertise
	icon = treaty
	level = 1
	conditions = {
		character_type = diplomat
		not = {
			trait = experienced_diplomat
			trait = masterful_diplomat
		}
	}
	modifier = {
		diplomacy = 1
	}
}
